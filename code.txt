import tkinter as tk
import random

# List of colors to choose from
colors = ["Red", "Blue", "Green", "Yellow", "Orange", "Purple"]

# Function to handle color prediction
def predict_color(user_choice):
    computer_choice = random.choice(colors)
    result_text.set(f"Computer chose: {computer_choice}")

    if user_choice == computer_choice:
        score[0] += 1
        result_label.config(fg="green")
        result_text.set(f"Correct! Computer chose: {computer_choice}")
    else:
        result_label.config(fg="red")
        result_text.set(f"Wrong! Computer chose: {computer_choice}")
    
    score_label.config(text=f"Score: {score[0]}")

# Create main window
root = tk.Tk()
root.title("Color Prediction Game")
root.geometry("400x300")
root.config(bg="white")

score = [0]

title_label = tk.Label(root, text="Guess the Color!", font=("Arial", 20), bg="white")
title_label.pack(pady=10)

button_frame = tk.Frame(root, bg="white")
button_frame.pack(pady=10)

# Create buttons for each color
for color in colors:
    btn = tk.Button(button_frame, text=color, width=10, bg=color.lower(), fg="white",
                    command=lambda c=color: predict_color(c))
    btn.pack(side="left", padx=5)

result_text = tk.StringVar()
result_label = tk.Label(root, textvariable=result_text, font=("Arial", 14), bg="white")
result_label.pack(pady=20)

score_label = tk.Label(root, text="Score: 0", font=("Arial", 16), bg="white")
score_label.pack()

root.mainloop()
